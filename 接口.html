Serial myPort; // 创建一个Serial对象
int sensorValue = 0; // 存储传感器的值

void setup() {
size(800, 600); // 设置画布大小
myPort = new Serial(this, Serial.list()[0], 9600); // 打开与Arduino的串行连接
}

void draw() {
if (myPort.available() > 0) { // 如果有可用的数据
sensorValue = myPort.read(); // 读取数据
}
background(0); // 清空画布
ellipse(width/2, height/2, sensorValue, sensorValue); // 根据传感器的值画一个圆形
